6.1.1 Write a Python program to check if a given date is valid. If the date is valid, print the next day's date (incremented date). If the date is invalid, print "Invalid Date".
# Read input
day = int(input())
month = int(input())
year = int(input())

# Function to check leap year
def is_leap(y):
    if (y % 400 == 0) or (y % 4 == 0 and y % 100 != 0):
        return True
    return False

# Validate year
if year <= 0:
    print("Invalid Date")

# Validate month
elif month < 1 or month > 12:
    print("Invalid Date")

else:
    # Days in each month
    if month in [1, 3, 5, 7, 8, 10, 12]:
        max_days = 31
    elif month in [4, 6, 9, 11]:
        max_days = 30
    elif month == 2:
        if is_leap(year):
            max_days = 29
        else:
            max_days = 28

    # Validate day
    if day < 1 or day > max_days:
        print("Invalid Date")
    else:
        # Increment date
        day += 1

        if day > max_days:
            day = 1
            month += 1

            if month > 12:
                month = 1
                year += 1

        # Print in DD-MM-YYYY format
        print(f"{day:02d}-{month:02d}-{year}")

6.1.2 Write a Python program to calculate the factorial of a number n using loops.
n = int(input())
factorial = 1
for i in range(1, n + 1):
    factorial *= i
print(factorial)
